Automate Competitor Monitoring in Python — The Complete Guide

By Mahika Jadhav · May 31, 2026 · 7 min read

Manual competitor research is slow, inconsistent, and expensive in human time. Automated monitoring solves consistency — but naive implementations trade human time for LLM spend, calling the model on every check whether or not anything changed. Here's how to build a complete automated competitor intelligence system that only fires the LLM on real changes.

What to monitor

A complete competitor monitoring setup watches multiple signal types:

Building the monitoring config

COMPETITORS = {
    "competitor-a": {
        "pricing": "https://competitor-a.com/pricing",
        "changelog": "https://competitor-a.com/changelog",
        "homepage": "https://competitor-a.com/",
        "blog": "https://competitor-a.com/blog",
    },
    "competitor-b": {
        "pricing": "https://competitor-b.com/plans",
        "homepage": "https://competitor-b.com/",
        "jobs": "https://competitor-b.com/careers",
    },
}

QUESTIONS = {
    "pricing": "Did pricing, tiers, or feature availability change? What specifically?",
    "changelog": "What new features, fixes, or deprecations were announced?",
    "homepage": "Did the main value proposition, headline, or CTA change?",
    "blog": "Were any new posts published? What are they about?",
    "jobs": "Were new roles posted? What functions are they hiring in?",
}

The monitoring loop

import indra
from anthropic import Anthropic

agent = indra.init(brightdata_api_key="your-bd-key")
client = Anthropic()

def analyze(goal, inputs, *_):
    content = inputs.get("diff") or inputs.get("content", "")
    return client.messages.create(
        model="claude-haiku-4-5", max_tokens=512,
        messages=[{"role": "user", "content": f"{goal}\n\n{content}"}]
    ).content[0].text

alerts = []

for company, pages in COMPETITORS.items():
    for page_type, url in pages.items():
        result = agent.watch(
            url=url,
            question=QUESTIONS.get(page_type, "What changed?"),
            generation_fn=analyze,
        )

        if result.changed:
            alerts.append({
                "company": company,
                "type": page_type,
                "url": url,
                "insight": result.insight,
            })

# Send all alerts in one digest
if alerts:
    send_competitive_digest(alerts)

# Log efficiency
stats = agent.get_stats()
total_pages = sum(len(p) for p in COMPETITORS.values())
print(f"Checked {stats['total_fetches']} pages, {stats['changes_detected']} changed")
print(f"LLM called {stats['llm_calls_fired']} times (vs {stats['total_fetches']} naive)")

Sending alerts to Slack

import requests

def send_competitive_digest(alerts):
    blocks = [{"type": "header", "text": {"type": "plain_text", "text": "🔍 Competitor Changes Detected"}}]

    for alert in alerts:
        blocks.append({
            "type": "section",
            "text": {
                "type": "mrkdwn",
                "text": f"*{alert['company'].title()}* — {alert['type']}\n{alert['insight']}\n<{alert['url']}|View page>"
            }
        })

    requests.post(SLACK_WEBHOOK_URL, json={"blocks": blocks})
