How to Monitor Website Changes with Python and an LLM

By Mahika Jadhav · May 31, 2026 · 7 min read

You want to know when a website changes — a competitor's product page, a regulatory filing, a supplier's lead times, a news feed. Python makes it easy to fetch pages. The challenge is doing it intelligently: not hammering the LLM with unchanged content, not getting blocked by anti-bot systems, and actually getting useful summaries of what changed.

Here's the complete guide.

The naive approach (and why it's expensive)

# Don't do this
import requests
from openai import OpenAI

client = OpenAI()

def check_for_changes(url):
    page = requests.get(url).text
    response = client.chat.completions.create(
        model="gpt-4o",
        messages=[{"role": "user", "content": f"What changed on this page?\n\n{page}"}]
    )
    return response.choices[0].message.content

# Runs every hour. Page hasn't changed in 6 days.
# You've paid for 144 LLM calls that all say "nothing changed."

The problem: you're asking the LLM to check for changes when the LLM doesn't know what it looked like before. And even if you provide the diff, you're still calling the LLM when there is no diff to provide.

The right approach: hash, compare, diff, then LLM

import hashlib, sqlite3, difflib, requests

def get_content_hash(url):
    return hashlib.sha256(requests.get(url).text.encode()).hexdigest()

def get_diff(old_content, new_content):
    return "\n".join(difflib.unified_diff(
        old_content.splitlines(), new_content.splitlines(),
        lineterm=""
    ))

# Step 1: fetch and hash — free
new_hash = get_content_hash(url)
old_hash = db.get_hash(url)

# Step 2: if unchanged, skip the LLM entirely
if new_hash == old_hash:
    return {"changed": False, "insight": cached_insight}

# Step 3: only if changed, build diff and call LLM
diff = get_diff(old_content, new_content)
insight = llm.analyze(diff)  # much cheaper — diff is tiny vs full page

Using Indra (handles all of this + bot detection)

Indra implements the hash-compare-diff-LLM pipeline, plus Bright Data integration for sites that block scrapers.

import indra
from anthropic import Anthropic

agent = indra.init(brightdata_api_key="your-bd-key")
client = Anthropic()

def analyze(goal, inputs, *_):
    content = inputs.get("diff") or inputs.get("content", "")
    return client.messages.create(
        model="claude-haiku-4-5", max_tokens=512,
        messages=[{"role": "user", "content": f"{goal}\n\n{content}"}]
    ).content[0].text

result = agent.watch(
    url="https://example.com/page-to-monitor",
    question="Summarize what changed and whether it's significant",
    generation_fn=analyze,
)

if result.changed:
    print(f"CHANGED: {result.insight}")
    print(f"Tokens used: {result.tokens_used}")
else:
    print(f"No change. Tokens saved: {result.tokens_saved}")

Monitoring multiple pages

pages = [
    {"url": "https://site1.com/docs/changelog", "question": "New features or breaking changes?"},
    {"url": "https://site2.com/pricing", "question": "Price or tier changes?"},
    {"url": "https://site3.com/status", "question": "Any new incidents or outages?"},
]

for page in pages:
    result = agent.watch(
        url=page["url"],
        question=page["question"],
        generation_fn=analyze,
    )
    if result.changed:
        notify(page["url"], result.insight)
